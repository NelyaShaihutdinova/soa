# Создание SSL конфигурации
create-ssl --certname server-cert --keyStore server-keystore.p12 --keyStorePassword changeit --keyStoreType PKCS12

# Отключение HTTP listener (который создается через --autobindhttp)
set configs.config.server-config.network-config.network-listeners.network-listener.http-listener-1.enabled=false

# Создание HTTPS listener на порту 8081
create-network-listener --protocol http-listener-1 --listenerport 8081 --enabled true --jkenabled false --target server-config ssl-listener-8081

# Привязка SSL к listener
set configs.config.server-config.network-config.network-listeners.network-listener.ssl-listener-8081.ssl.server-cert-nickname=server-cert
